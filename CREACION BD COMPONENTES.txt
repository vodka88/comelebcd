//INICIALIZAR TABLAS
TRUNCATE TABLE  EMPAQUETADO
TRUNCATE TABLE MOVIMIENTOS



create table PRINCIPAL
(
    CLAVE VARCHAR2(6) not null,
    COMPONENTE VARCHAR2(40),
    CLASE VARCHAR2(6),
    DATASHEET VARCHAR2(6),
    CANTIDAD VARCHAR2(6),
    ESTADO VARCHAR2(2),
    FECHA VARCHAR2(10),
    primary key (CLAVE)
)

create table CLASE
(
    CLAVE VARCHAR2(6) not null,
    CLASE VARCHAR2(20),
    TIPO VARCHAR2(20),
    SUBTIPO VARCHAR2(20),
    EMPAQUETADO VARCHAR2(4),
    FECHA VARCHAR2(10),
    primary key (CLAVE)
)

create table MOVIMIENTOS
(
    CLAVE VARCHAR2(6) not null,
    CODTABLA VARCHAR2(2),
    VALMOD VARCHAR2(72),
    FECHA VARCHAR2(10),
    TIPOMOD VARCHAR2(2),
    DESCRIPCION VARCHAR2(20),
    primary key (CLAVE)
)

create table EMPAQUETADO
(
    CLAVE VARCHAR2(6) not null,
    ENCAPSULADO VARCHAR2(8),
    NUMPATILLAS VARCHAR2(4),
    FECHA VARCHAR2(10),
    primary key (CLAVE)
)

create table ESTADO
(
    CLAVE VARCHAR2(6) not null,
    DESESTADO VARCHAR2(8),
    FECHA VARCHAR2(10),
    primary key (CLAVE)
)

create table DATASHEET
(
    CLAVE VARCHAR2(6) not null,
    CODTABLAPRIM VARCHAR2(6),
    URL VARCHAR2(80),
    DATASHEET VARCHAR2(80),
    FECHA VARCHAR2(10),
    primary key (CLAVE)
)
